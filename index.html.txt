    // Monitorização das estatísticas administrativas
    onValue(window.dbRef, (snapshot) => {

      if (data) {

        const total = campingsList.length;
        const sumScore = campingsList.reduce((acc, curr) => acc + (parseFloat(curr.nota) || 0), 0);
        const avgScore = total > 0 ? (sumScore / total).toFixed(1) : "0.0";

        document.getElementById('stat-score').innerText = avgScore;

      const city = document.getElementById('cidade').value.trim();
      const state = document.getElementById('estado').value.trim().toUpperCase();
      const foto = document.getElementById('foto').value.trim() || '[https://images.unsplash.com/photo-1504280390367-361c6d9f38f4?auto=format&fit=crop&w=500&q=80](https://images.unsplash.com/photo-1504280390367-361c6d9f38f4?auto=format&fit=crop&w=500&q=80)';
      const preco = parseFloat(document.getElementById('preco').value) || 0;

      const campingData = {

        estado: state,

        preco,
        nota,

        banheiro_limpeza: getGroupSelectionText('banheiro_limpeza'),

      };

          console.error(err);
        });

  <script>
    function selectBtn(btn) {
      const parent = btn.parentElement;
      const buttons = parent.querySelectorAll('.option-btn');
      buttons.forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      calcularNotaTecnica();
    }

    function getGroupSelectionText(groupName) {
      const groupContainer = document.querySelector(`[data-group="${groupName}"]`);
      if (groupContainer) {
        const activeBtn = groupContainer.querySelector('.option-btn.active');
        return activeBtn ? activeBtn.innerText : "Não Avaliado";
      }
      return "N/A";
    }

    function calcularNotaTecnica() {
      const pesos = {
        energia: { peso: 0.25, grupos: ['energia_tensao', 'energia_estabilidade', 'energia_amperagem'] },
        banheiro: { peso: 0.20, grupos: ['banheiro_seco', 'banheiro_proporcao', 'banheiro_quente', 'banheiro_limpeza'] },
        acesso: { peso: 0.20, grupos: ['acesso_estrada', 'acesso_trailer', 'acesso_nivel'] },
        conectividade: { peso: 0.10, grupos: ['conec_sinal', 'conec_wifi'] },
        cozinha: { peso: 0.15, grupos: ['cozinha_geral'] },
        infra: { peso: 0.10, grupos: ['infra_lixo', 'infra_seguranca'] }
      };

      let notaFinal = 0;

      for (const [key, secao] of Object.entries(pesos)) {
        let somaSecao = 0;
        let countAtivos = 0;

        secao.grupos.forEach(grupo => {
          const container = document.querySelector(`[data-group="${grupo}"]`);
          if (container) {
            const activeBtn = container.querySelector('.option-btn.active');
            if (activeBtn) {
              somaSecao += parseFloat(activeBtn.getAttribute('data-value')) || 0;
              countAtivos++;
            }
          }
        });

        if (countAtivos > 0) {
          const mediaSecao = (somaSecao / countAtivos);
          const notaSecaoDeDez = (mediaSecao / 3) * 10;
          notaFinal += notaSecaoDeDez * secao.peso;
        }
      }

      const finalClamped = Math.min(10.0, Math.max(0.0, notaFinal));
      document.getElementById('nota-display').innerText = finalClamped.toFixed(1);
    }

    function showToast(message, classes) {
      const toast = document.getElementById('toast');
      const toastText = document.getElementById('toast-text');
      toast.className = `mb-4 p-3 rounded-xl border text-sm flex items-center justify-between ${classes}`;
      toastText.innerText = message;
      toast.classList.remove('hidden');
      setTimeout(() => {
        toast.classList.add('hidden');
      }, 4000);
    }

    function limparFormulario() {
      document.getElementById('campingForm').reset();
      const activeButtons = document.querySelectorAll('.option-btn.active');
      activeButtons.forEach(b => b.classList.remove('active'));
      document.getElementById('nota-display').innerText = "0.0";
    }
  </script>
