  <script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-app.js";
    import { getDatabase, ref, onValue } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-database.js";

    // Modifique com as suas credenciais originais do console Firebase
    const firebaseConfig = {
      apiKey: "COLE_AQUI_SUA_API_KEY",
      authDomain: "COLE_AQUI_SEU_AUTH_DOMAIN",
      databaseURL: "COLE_AQUI_SUA_DATABASE_URL",
      projectId: "COLE_AQUI_SEU_PROJECT_ID",
      storageBucket: "COLE_AQUI_SEU_STORAGE_BUCKET",
      messagingSenderId: "COLE_AQUI_SEU_SENDER_ID",
      appId: "COLE_AQUI_SEU_APP_ID"
    };

    const app = initializeApp(firebaseConfig);
    const db = getDatabase(app);
    const dbRef = ref(db, 'campings');

    window.campingsCache = [];
    window.campingSelecionado = null;

    // Escuta ativa em tempo real
    onValue(dbRef, (snapshot) => {
      const data = snapshot.val();
      if (data) {
        window.campingsCache = Object.keys(data).map(key => ({
          id: key,
          ...data[key]
        }));
        
        atualizarContadores();
        renderizarInicio();
        renderizarRanking();
      } else {
        document.getElementById('lista-campings').innerHTML = `
          <div class="text-center py-12 text-gray-500">Nenhum camping registado até ao momento.</div>
        `;
        document.getElementById('ranking-lista').innerHTML = `
          <div class="text-center py-6 text-gray-500">Sem classificação disponível de momento.</div>
        `;
      }
    });

    function atualizarContadores() {
      const campings = window.campingsCache;
      const total = campings.length;
      const sum = campings.reduce((acc, curr) => acc + (parseFloat(curr.nota) || 0), 0);
      const media = total > 0 ? (sum / total).toFixed(1) : "0.0";
      const estadosUnicos = new Set(campings.map(c => c.estado?.toUpperCase())).size;

      document.getElementById('stat-campings').innerText = total;
      document.getElementById('stat-score').innerText = media;
      document.getElementById('stat-estados').innerText = estadosUnicos;
    }

  <script>
    function mudarTela(telaId) {
      document.getElementById('tela-inicio').classList.add('hidden');
      document.getElementById('tela-ficha').classList.add('hidden');
      document.getElementById('tela-ranking').classList.add('hidden');

      document.getElementById('btn-inicio').classList.remove('nav-active');
      document.getElementById('btn-ficha').classList.remove('nav-active');
      document.getElementById('btn-ranking').classList.remove('nav-active');

      document.getElementById(`tela-${telaId}`).classList.remove('hidden');
      document.getElementById(`btn-${telaId}`).classList.add('nav-active');

      window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    function renderizarInicio(filtrados = null) {
      const items = filtrados || window.campingsCache;
      const container = document.getElementById('lista-campings');

      if (items.length === 0) {
        container.innerHTML = `<div class="text-center py-8 text-gray-500">Nenhum resultado encontrado para a pesquisa.</div>`;
        return;
      }

      container.innerHTML = items.map(c => `
        <div class="bg-white rounded-2xl overflow-hidden shadow-sm hover:shadow-md transition-all cursor-pointer flex flex-col" onclick="abrirFicha('${c.id}')">
          <div class="h-44 w-full bg-cover bg-center relative" style="background-image: url('${c.foto}')">
            <div class="absolute top-3 right-3 bg-[#1b3d2f] text-[#d4af37] font-serif font-bold text-sm px-3 py-1 rounded-full shadow-md">
              ★ ${parseFloat(c.nota).toFixed(1)}
            </div>
          </div>
          <div class="p-4 flex justify-between items-center bg-white">
            <div>
              <h3 class="font-serif font-bold text-brand-dark text-lg">${c.nome}</h3>
              <p class="text-xs text-gray-500 flex items-center gap-1">📍 ${c.cidade} - ${c.estado}</p>
            </div>
            <div class="text-right">
              <span class="block text-[10px] text-gray-400 uppercase tracking-widest">Diária aprox.</span>
              <span class="text-sm font-bold text-emerald-800">R$ ${c.preco ? parseFloat(c.preco).toFixed(2) : '---'}</span>
            </div>
          </div>
        </div>
      `).join('');
    }

    function filtrarCampings() {
      const valor = document.getElementById('busca-input').value.toLowerCase().trim();
      if (!valor) {
        renderizarInicio();
        return;
      }

      const filtrados = window.campingsCache.filter(c => 
        c.nome.toLowerCase().includes(valor) || 
        c.cidade.toLowerCase().includes(valor) || 
        c.estado.toLowerCase().includes(valor)
      );
      renderizarInicio(filtrados);
    }

    function renderizarRanking() {
      const ordenados = [...window.campingsCache].sort((a, b) => (b.nota || 0) - (a.nota || 0));
      const container = document.getElementById('ranking-lista');

      container.innerHTML = ordenados.map((c, index) => {
        let podioEstilo = 'text-gray-500 bg-gray-100';
        if (index === 0) podioEstilo = 'bg-amber-100 text-amber-700 font-bold';
        if (index === 1) podioEstilo = 'bg-slate-200 text-slate-700 font-bold';
        if (index === 2) podioEstilo = 'bg-orange-100 text-orange-700 font-bold';

        return `
          <div class="flex items-center gap-3 p-3 hover:bg-gray-50 rounded-xl transition-all border-b last:border-0 cursor-pointer" onclick="abrirFicha('${c.id}')">
            <span class="w-8 h-8 rounded-full flex items-center justify-center text-xs shrink-0 ${podioEstilo}">
              ${index + 1}º
            </span>
            <div class="flex-1 min-w-0">
              <h4 class="font-bold text-brand-dark text-sm truncate">${c.nome}</h4>
              <p class="text-[10px] text-gray-400 truncate">${c.cidade} (${c.estado})</p>
            </div>
            <div class="text-right shrink-0">
              <span class="text-sm font-serif font-bold text-brand-dark">${parseFloat(c.nota).toFixed(1)}</span>
            </div>
          </div>
        `;
      }).join('');
    }

    function abrirFicha(id) {
      const camping = window.campingsCache.find(c => c.id === id);
      if (!camping) return;

      window.campingSelecionado = camping;

      const container = document.getElementById('detalhe-container');
      container.innerHTML = `
        <div class="bg-white rounded-3xl overflow-hidden custom-shadow">
          <div class="h-56 w-full bg-cover bg-center relative" style="background-image: url('${camping.foto}')">
            <div class="absolute inset-0 bg-gradient-to-t from-black/80 via-transparent to-transparent"></div>
            
            <div class="absolute bottom-4 left-4 right-4 text-white">
              <span class="text-xs font-serif text-brand-gold uppercase tracking-wider block">${camping.cidade} - ${camping.estado}</span>
              <h2 class="text-xl font-serif font-bold tracking-wide">${camping.nome}</h2>
            </div>

            <div class="absolute top-4 right-4 bg-[#1b3d2f] text-[#d4af37] font-serif font-bold text-sm px-4 py-1.5 rounded-full shadow-lg">
              ★ ${parseFloat(camping.nota).toFixed(1)}
            </div>
          </div>

          <div class="p-6 space-y-6">
            <div class="flex justify-between items-center border-b pb-4">
              <div>
                <span class="text-xs text-gray-400 block uppercase">Diária estimada</span>
                <span class="text-lg font-bold text-emerald-800">R$ ${camping.preco ? parseFloat(camping.preco).toFixed(2) : 'Sob consulta'}</span>
              </div>
              <div class="bg-emerald-50 text-emerald-800 text-xs px-3 py-1.5 rounded-full font-bold">
                ✓ Verificado
              </div>
            </div>

            <div class="space-y-4">
              <h3 class="text-xs font-serif uppercase tracking-widest text-brand-dark font-bold border-b pb-1">Ficha Técnica Detalhada</h3>
              
              <!-- Seção de Energia -->
              <div class="bg-gray-50 rounded-xl p-3 space-y-2">
                <span class="text-xs font-bold text-[#1b3d2f] flex items-center gap-1">⚡ Energia Elétrica</span>
                <div class="grid grid-cols-2 gap-2 text-xs text-gray-600">
                  <div>Tensão: <span class="font-semibold text-brand-dark">${camping.energia_tensao || 'N/A'}</span></div>
                  <div>Estabilidade: <span class="font-semibold text-brand-dark">${camping.energia_estabilidade || 'N/A'}</span></div>
                  <div class="col-span-2">Amperagem: <span class="font-semibold text-brand-dark">${camping.energia_amperagem || 'N/A'}</span></div>
                </div>
              </div>

              <!-- Seção de Banheiros -->
              <div class="bg-gray-50 rounded-xl p-3 space-y-2">
                <span class="text-xs font-bold text-[#1b3d2f] flex items-center gap-1">🚿 Banheiros & Chuveiro</span>
                <div class="grid grid-cols-2 gap-2 text-xs text-gray-600">
                  <div>Área Seca: <span class="font-semibold text-brand-dark">${camping.banheiro_seco || 'N/A'}</span></div>
                  <div>Proporção: <span class="font-semibold text-brand-dark">${camping.banheiro_proporcao || 'N/A'}</span></div>
                  <div>Chuveiro: <span class="font-semibold text-brand-dark">${camping.banheiro_quente || 'N/A'}</span></div>
                  <div>Limpeza: <span class="font-semibold text-brand-dark">${camping.banheiro_limpeza || 'N/A'}</span></div>
                </div>
              </div>

              <!-- Seção de Terreno e Acesso -->
              <div class="bg-gray-50 rounded-xl p-3 space-y-2">
                <span class="text-xs font-bold text-[#1b3d2f] flex items-center gap-1">🚗 Estrada e Acesso</span>
                <div class="grid grid-cols-2 gap-2 text-xs text-gray-600">
                  <div>Estrada: <span class="font-semibold text-brand-dark">${camping.acesso_estrada || 'N/A'}</span></div>
                  <div>Trailer: <span class="font-semibold text-brand-dark">${camping.acesso_trailer || 'N/A'}</span></div>
                  <div class="col-span-2">Nível Terreno: <span class="font-semibold text-brand-dark">${camping.acesso_nivel || 'N/A'}</span></div>
                </div>
              </div>

              <!-- Seção de Conectividade -->
              <div class="bg-gray-50 rounded-xl p-3 space-y-2">
                <span class="text-xs font-bold text-[#1b3d2f] flex items-center gap-1">📶 Conectividade</span>
                <div class="grid grid-cols-2 gap-2 text-xs text-gray-600">
                  <div>Sinal Celular: <span class="font-semibold text-brand-dark">${camping.conec_sinal || 'N/A'}</span></div>
                  <div>Wi-Fi Local: <span class="font-semibold text-brand-dark">${camping.conec_wifi || 'N/A'}</span></div>

              <!-- Cozinha e Serviços -->
              <div class="bg-gray-50 rounded-xl p-3 space-y-2">
                <span class="text-xs font-bold text-[#1b3d2f] flex items-center gap-1">🍳 Cozinha & Serviços</span>

      mudarTela('ficha');
